  64 | def trigger_hook(event: str, *args: Any, **kwargs: Any) -> str | None:
  65 |     """Run handlers in registration order and return the first blocking result."""
  66 |     for handler in HOOKS.get(event, []):

  90 | def permission_hook(block: Any) -> str | None:
  91 |     name = block_attr(block, "name")
  92 |     tool_input = block_attr(block, "input", {}) or {}
  93 | 
  94 |     if name == "bash":
  95 |         command = str(tool_input.get("command", ""))
  96 |         if matches_any(command, DENY_PATTERNS):
  97 |             STATS["blocked"] += 1
  98 |             append_log(f"blocked name=bash reason=deny command={command!r}")
  99 |             return f"BLOCKED: dangerous command denied: {command}"
 100 |         if matches_any(command, RISKY_PATTERNS) and not approve(
 101 |             f"Allow risky command? {command}"
 102 |         ):
 103 |             STATS["blocked"] += 1
 104 |             append_log(f"blocked name=bash reason=user_denied command={command!r}")
 105 |             return f"BLOCKED: user denied command: {command}"
 106 | 
 107 |     if name in {"write_file", "edit_file", "read_file"}:
 108 |         path = str(tool_input.get("path", ""))
 109 |         if path and path_escapes_workspace(path):
 110 |             STATS["blocked"] += 1
 111 |             append_log(f"blocked name={name} reason=path_escape path={path!r}")
 112 |             return f"BLOCKED: path escapes workspace: {path}"
 113 | 
 114 |     return None
