  65 | def trigger_hook(event: str, *args: Any, **kwargs: Any) -> str | None:
  66 |     """Run handlers and return the first non-None result.
  67 | 
  68 |     PreToolUse and PostToolUse automatically emit internal OnToolStart and
  69 |     OnToolEnd events, so business code gets structured tool logs for free.
  70 |     """
  71 |     if event == "PreToolUse" and args:
  72 |         trigger_hook("OnToolStart", args[0])
  73 |     elif event == "PostToolUse" and len(args) >= 2:
  74 |         trigger_hook("OnToolEnd", args[0], args[1])
  75 | 
  76 |     for handler in HOOKS.get(event, []):
  77 |         result = handler(*args, **kwargs)
  78 |         if result is not None:
  79 |             return result
  80 |     return None

  90 |     return any(re.search(pattern, command, flags=re.IGNORECASE) for pattern in patterns)

  93 | def permission_hook(block: Any) -> str | None:
  94 |     name = block_attr(block, "name")
  95 |     tool_input = block_attr(block, "input", {}) or {}
  96 | 
  97 |     if name == "bash":
  98 |         command = str(tool_input.get("command", ""))
  99 |         if matches_any(command, DENY_PATTERNS):
 100 |             STATS["blocked"] += 1
 101 |             log_event("ERROR", "blocked", name="bash", reason="deny", command=command)
 102 |             return f"BLOCKED: dangerous command denied: {command}"
 103 |         if matches_any(command, RISKY_PATTERNS) and not approve(
 104 |             f"Allow risky command? {command}"
 105 |         ):
 106 |             STATS["blocked"] += 1
 107 |             log_event("WARN", "blocked", name="bash", reason="user_denied", command=command)
 108 |             return f"BLOCKED: user denied command: {command}"
 109 | 
 110 |     if name in {"write_file", "edit_file", "read_file"}:
 111 |         path = str(tool_input.get("path", ""))
 112 |         if path and path_escapes_workspace(path):
 113 |             STATS["blocked"] += 1
 114 |             log_event("ERROR", "blocked", name=name, reason="path_escape", path=path)
 115 |             return f"BLOCKED: path escapes workspace: {path}"
 116 | 
 117 |     return None

 151 | def on_tool_start(block: Any) -> str | None:
 152 |     name = block_attr(block, "name", "<unknown>")
 153 |     tool_id = block_attr(block, "id", "<no-id>")
 154 |     tool_input = block_attr(block, "input", {}) or {}
 155 |     TOOL_STARTS[tool_id] = time.perf_counter()
 156 |     STATS["tool_calls"] += 1
 157 |     if name == "spawn_subagent":
 158 |         STATS["subagents"] += 1
 159 |     category = "SUBAGENT" if name == "spawn_subagent" else "TOOL"
 160 |     log_event(category, "start", name=name, id=tool_id, input=compact_input(tool_input))
 161 |     return None
 162 | 
 163 | 
 164 | def on_tool_end(block: Any, output: str) -> str | None:
 165 |     name = block_attr(block, "name", "<unknown>")
 166 |     tool_id = block_attr(block, "id", "<no-id>")
 167 |     start = TOOL_STARTS.pop(tool_id, None)
 168 |     elapsed_ms = None if start is None else int((time.perf_counter() - start) * 1000)
 169 |     category = "SUBAGENT" if name == "spawn_subagent" else "TOOL"
 170 |     log_event(category, "end", name=name, id=tool_id, chars=len(output), elapsed_ms=elapsed_ms)
 171 |     return None

 174 | def approve(question: str) -> bool:
 175 |     if AUTO_APPROVE:
 176 |         return True
 177 |     answer = input(f"{question}\n[y/N] ").strip().lower()
 178 |     return answer in {"y", "yes"}

 182 |     rendered = repr(tool_input)
