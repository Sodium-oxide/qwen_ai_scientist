 108 | def run_agent(user_input: str) -> str:
 109 |     client = get_client()
 110 |     log_event("USER", "prompt", chars=len(user_input))
 111 |     injected = trigger_hook("UserPromptSubmit", user_input)
 112 |     prompt = injected if injected is not None else user_input
 113 |     messages: list[dict[str, Any]] = [{"role": "user", "content": prompt}]
 114 | 
 115 |     while True:
 116 |         messages[:] = compact_messages(messages)
 117 |         response = create_response(client, user_input, messages)
 118 | 
 119 |         if response.stop_reason == "end_turn":
 120 |             final_text = response_text(response.content)
 121 |             stop_injection = trigger_hook("Stop", final_text)
 122 |             if stop_injection is not None:
 123 |                 messages.append({"role": "user", "content": stop_injection})
 124 |                 continue
 125 |             log_event("AGENT", "final", text=final_text)
 126 |             return final_text
 127 | 
 128 |         tool_blocks = [
 129 |             block for block in response.content if block_attr(block, "type") == "tool_use"
 130 |         ]
 131 |         if not tool_blocks:
 132 |             final_text = response_text(response.content)
 133 |             trigger_hook("Stop", final_text)
 134 |             log_event("AGENT", "final", text=final_text)
 135 |             return final_text
 136 | 
 137 |         messages.append(
 138 |             {
 139 |                 "role": "assistant",
 140 |                 "content": [block_to_dict(block) for block in response.content],
 141 |             }
 142 |         )
 143 |         messages.append(
 144 |             {
 145 |                 "role": "user",
 146 |                 "content": [run_tool(block, messages) for block in tool_blocks],
 147 |             }
